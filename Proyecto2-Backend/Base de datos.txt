CREATE TABLE hoteles(
	id_hotel VARCHAR(20) PRIMARY KEY,
	nombre VARCHAR(50),
	ciudad  VARCHAR(50),
	direccion VARCHAR(50),
	imagen_url VARCHAR(255)
);


CREATE TABLE tipo_habitacion(
	id_tipo VARCHAR(10) PRIMARY KEY,
	tipo VARCHAR(10),
	descripcion VARCHAR(30),
	precio INT,
    capacidad_min INT,
    capacidad_max INT
);


CREATE TABLE habitacion(
	id_habitacion VARCHAR(20) PRIMARY KEY,
	num_habitacion INT,
	id_hotel VARCHAR(20),
	id_tipo VARCHAR(10),
	FOREIGN KEY (id_hotel) REFERENCES hoteles(id_hotel),
	FOREIGN KEY (id_tipo) REFERENCES tipo_habitacion(id_tipo)
);


CREATE TABLE reservas(
	id_reserva INT AUTO_INCREMENT PRIMARY KEY,
	fecha_llegada DATE,
	fecha_salida DATE,
	id_hotel VARCHAR(20),
	id_habitacion VARCHAR(20),
	nombre_cliente VARCHAR(50),
	apellido_cliente VARCHAR(50),
    telefono_cliente VARCHAR(20),
	correo_cliente VARCHAR(20),
	cant_personas INT,
	FOREIGN KEY (id_hotel) REFERENCES hoteles(id_hotel),
	FOREIGN KEY (id_/habitacion) REFERENCES habitacion(id_habitacion)
)
